%matplotlib inline
import numpy as np
import matplotlib.pyplot as plt
import scipy.fftpack

# Number of samplepoints
N = 600
# sample spacing
T = 1.0 / 800.0
x = np.linspace(0.0, N*T, N)
y = 0*x
a = np.array([1.0, 0.5, 0.6, 0.3])
freq = np.array([10.0, 50.0, 1e2])
for i in range(0,freq.size):
    y += a[i]*np.sin(freq[i] * 2.0*np.pi*x)
yf = scipy.fftpack.fft(y) # FFT
xf = np.linspace(0.0, 1.0/(2.0*T), N/2)

plt.rcParams['figure.figsize'] = 2*6.675, 4.181

plt.subplot(1,2,1)
fac = int(0.1*N)
plt.plot(x[1:fac], y[1:fac])
plt.grid(True)
plt.subplot(1,2,2)
plt.plot(xf, 2.0/N * np.abs(yf[:N//2]))
plt.grid(True)

plt.draw()
plt.savefig("FFT_exemplo.png")
plt.show()

yf_limpo = yf

Acima, foi adicionado ruído. Esse ruído adiciona ganhos em todas as frequências. Das baixas até às altas. Um sinal composto por contribuições em todas as frequências se chama ruído branco (uma alusão à luz branca, formada de feixes de várias frequências no espectro do visível). Um sinal aleatório com PDF gaussiana é um ruído branco gaussiano. Um sinal aleatório com uma faixa de frequências é chamado ruído cinza.¶

def butter_lowpass(cutoff, fs, order=5):
    nyq = 0.5 * fs
    normal_cutoff = cutoff / nyq
    b, a = butter(order, normal_cutoff, btype='low', analog=False)
    return b, a


def butter_lowpass_filter(data, cutoff, fs, order=5):
    b, a = butter_lowpass(cutoff, fs, order=order)
    y = lfilter(b, a, data)
    return y


# Filter requirements.
